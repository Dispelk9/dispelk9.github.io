AWS

Initiate AWS CLI

export AWS_ACCESS_KEY_ID=
export AWS_SECRET_ACCESS_KEY=
export AWS_SESSION_TOKEN=

Checking which RDS using which Secret in Secret Manager
aws rds help
aws rds describe-db-clusters
aws rds describe-db-clusters --db-cluster-identifier --region eu-central-1 --query 'DBClusters[0].DatabaseName' --output text

Create S3
aws s3 help
aws s3api create-bucket --bucket bucket_name --region eu-central-1 --create-bucket-configuration LocationConstraint=eu-central-1

Get secret values from secretsmanager
aws secretsmanager get-secret-value --secret-id 'rds!cluster-05149d24-558b-4ca6-b696-d568646a31ef' --region eu-central-1 --query SecretString --output text
aws secretsmanager get-secret-value --secret-id 'rds!cluster-05149d24-558b-4ca6-b696-d568646a31ef' --query SecretString --output text | jq -r
